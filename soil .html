1️⃣

Sense

A capacitive soil moisture sensor reads the water content in your plant's soil every 60 seconds and sends an analog voltage to the ESP32's ADC pin.

2️⃣

Decide

The ESP32 compares the reading against thresholds. If the soil is too dry, it triggers the relay which activates the water pump for 3 seconds.

3️⃣

Report

Over WiFi, the ESP32 POSTs sensor data to your Node.js backend, which stores readings in a SQLite database and serves them to a dashboard.

4️⃣

Notify

When the pump fires or moisture is critically low, your backend sends an instant push notification to your phone via ntfy.sh — no app building required.

Hardware Components

Everything you need — total cost around $20–25 from Amazon or AliExpress.

01

ESP32 Development Board

The brains. Has built-in WiFi + Bluetooth, dual-core processor, and multiple ADC pins for analog sensor reading. Much better than Arduino for IoT.

~$8 REQUIRED
02

Capacitive Soil Moisture Sensor v1.2

Measures soil moisture via capacitance (not resistance), so it won't corrode. Outputs 0–3.3V analog signal. Much more durable than resistive sensors.

~$3 REQUIRED
03

5V Single-Channel Relay Module

Acts as a switch between the ESP32 (3.3V logic) and the pump (5V power). Active-LOW design means GPIO LOW = pump ON.

~$2 REQUIRED
04

Mini Submersible Water Pump (3–5V)

Small DC pump that draws water from a reservoir and delivers it to the plant. Runs for 2–5 seconds to deliver ~10–20ml per watering cycle.

~$3 REQUIRED
05

Jumper Wires + Breadboard

For prototyping without soldering. Male-to-female wires for sensor/relay, male-to-male for breadboard connections. Upgrade to perfboard later.

~$5 REQUIRED
06

USB Power Supply + Tubing

5V 2A USB adapter for power. Silicone tubing (5mm inner diameter) to route water from reservoir to plant. Buy 1–2 meters.

~$4 REQUIRED

ESP32 Firmware

Write this in Arduino IDE. Install board: "esp32 by Espressif". Required libraries: ArduinoJson, HTTPClient (built-in).

01

Install Arduino IDE + ESP32 Board

Download Arduino IDE 2.x. Go to File → Preferences, paste this URL in "Additional Boards Manager URLs":
https://raw.githubusercontent.com/espressif/arduino-esp32/gh-pages/package_esp32_index.json
Then go to Tools → Board → Boards Manager, search "esp32" and install.

02

Calibrate Your Sensor

Before writing the full firmware, upload this quick sketch to find your sensor's dry/wet values — they vary by sensor.

calibrate.ino — Run this first
#define MOISTURE_PIN 34

void setup() {
  Serial.begin(115200);
}

void loop() {
  int raw = analogRead(MOISTURE_PIN);
  // Stick sensor in dry air → note this value (your DRY_THRESHOLD)
  // Submerge in water → note this value (your WET_THRESHOLD)
  Serial.print("Raw ADC: "); Serial.println(raw);
  delay(1000);
}
03

Main Firmware — Copy & Customize

Replace WiFi credentials, server URL, and thresholds from your calibration above.

main.ino — Full firmware
#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>

// ── CONFIG ─────────────────────────────────────
#define MOISTURE_PIN  34
#define RELAY_PIN     26
#define PUMP_ON       LOW   // Relay is active-LOW
#define PUMP_OFF      HIGH
#define DRY_THRESHOLD 2800  // Tune from calibration!
#define WET_THRESHOLD 1500  // Tune from calibration!
#define PUMP_DURATION 3000  // ms to run pump
#define CHECK_INTERVAL 60000 // ms between readings

const char* SSID      = "YOUR_WIFI_NAME";
const char* PASSWORD  = "YOUR_WIFI_PASS";
const char* SERVER_URL = "http://192.168.1.100:3000/api/reading";
// ───────────────────────────────────────────────

void setup() {
  Serial.begin(115200);
  pinMode(RELAY_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, PUMP_OFF);

  WiFi.begin(SSID, PASSWORD);
  Serial.print("Connecting to WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500); Serial.print(".");
  }
  Serial.println("\nConnected! IP: " + WiFi.localIP().toString());
}

void loop() {
  int raw         = analogRead(MOISTURE_PIN);
  int moisturePct = map(raw, 4095, 0, 0, 100);
  bool pumped     = false;

  Serial.printf("Moisture: %d%% (raw: %d)\n", moisturePct, raw);

  if (raw > DRY_THRESHOLD) {
    Serial.println("Soil dry → activating pump...");
    digitalWrite(RELAY_PIN, PUMP_ON);
    delay(PUMP_DURATION);
    digitalWrite(RELAY_PIN, PUMP_OFF);
    pumped = true;
    Serial.println("Pump OFF.");
  }

  sendReading(moisturePct, pumped);
  delay(CHECK_INTERVAL);
}

void sendReading(int moisture, bool pumped) {
  if (WiFi.status() != WL_CONNECTED) return;
  HTTPClient http;
  http.begin(SERVER_URL);
  http.addHeader("Content-Type", "application/json");

  StaticJsonDocument<128> doc;
  doc["moisture"]   = moisture;
  doc["pump_fired"] = pumped;
  doc["device_id"]  = "plant-01";

  String body; serializeJson(doc, body);
  int code = http.POST(body);
  Serial.printf("Server responded: %d\n", code);
  http.end();
}

03

Write the Server

Copy this into server.js and change the ntfy topic to your unique name.

server.js — Full backend
const express = require('express');
const Database = require('better-sqlite3');
const axios = require('axios');

const app = express();
app.use(express.json());
app.use(express.static('public')); // Serve dashboard files

// ── DATABASE SETUP ──────────────────────────────
const db = new Database('plant.db');
db.exec(`
  CREATE TABLE IF NOT EXISTS readings (
    id        INTEGER PRIMARY KEY AUTOINCREMENT,
    device_id TEXT,
    moisture  INTEGER,
    pump_fired INTEGER,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
  )
`);

// ── NTFY NOTIFICATION ───────────────────────────
const NTFY_TOPIC = 'my-plant-abc123'; // CHANGE THIS

async function notify(title, body, priority = 'default') {
  try {
    await axios.post(`https://ntfy.sh/${NTFY_TOPIC}`, body, {
      headers: { 'Title': title, 'Priority': priority }
    });
  } catch (e) { console.error('Notification failed:', e.message); }
}

// ── RECEIVE DATA FROM ESP32 ─────────────────────
app.post('/api/reading', async (req, res) => {
  const { moisture, pump_fired, device_id } = req.body;

  // Save to database
  db.prepare(`
    INSERT INTO readings (device_id, moisture, pump_fired)
    VALUES (?, ?, ?)
  `).run(device_id, moisture, pump_fired ? 1 : 0);

  console.log(`[${device_id}] Moisture: ${moisture}% | Pumped: ${pump_fired}`);

  // Send notifications
  if (pump_fired) {
    await notify(
      '🌱 Plant Watered',
      `Moisture was low (${moisture}%). Auto-watering triggered.`
    );
  }
  if (moisture < 15) {
    await notify(
      '⚠️ Critical: Plant Needs Water',
      `Moisture critically low at ${moisture}%! Check your system.`,
      'urgent'
    );
  }
  res.json({ status: 'ok' });
});

// ── SERVE READINGS TO DASHBOARD ─────────────────
app.get('/api/readings', (req, res) => {
  const rows = db.prepare(`
    SELECT * FROM readings ORDER BY created_at DESC LIMIT 100
  `).all();
  res.json(rows.reverse()); // Oldest first for chart
});

app.listen(3000, () => {
  console.log('🌿 Plant server running on http://localhost:3000');
});
